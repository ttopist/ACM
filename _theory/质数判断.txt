uint32_t powmod (uint32_t base, uint32_t exp, uint32_t mod)
{
    uint32_t res = 1;
    base %= mod;
    while (exp) {
        if (exp % 2)
            res = res * (uint64_t)base % mod;
        exp /= 2;
        base = base * (uint64_t)base % mod;
    }
    return res;
}
